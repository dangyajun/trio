\section{Scanning}

The scanning is controlled by the format string.
The format string can contain normal text and conversion indicators.
The normal text can be any character except the nil character
(ASCII 000 = '\Bs 0') and the percent character (ASCII 045 = '\%').
Conversion indicators consists of an indication character (\%), followed by
zero or more conversion modifiers, and exactly one conversion specifier.


\subsection{Modifiers}

\begin{description}
\item Positional (9\$) [UNIX98]\\
See \Cmd{trio_printf}{3}.

\subsection{Specifiers}

\begin{description}
\item Percent (\%)\\

\item Character (c)\\

\item Decimal (d)\\

\item Floating-point (a, A, e, E, f, F, g, G)\\

\item Integer (i)\\

\item Count (n)\\

\item Octal (o)\\

\item Pointer (p)\\

\item String (s)\\

\item Unsigned (u)\\

\item Hex (x, X)\\

\item Scanlist ([])\\

\begin{description}
\item Exclusion (\Circum)\\

\item Range (-) [TRIO]\\
\begin{itemize}
\item Only increasing ranges, i.e. [a-b], but not [b-a].
\item Transitive ranges, ie. [a-b-c] equals [a-c].
\item Trailing minus, ie. [a-] is interpreted as an 'a' and a '-'.
\item Duplicates are ignored.
\end{itemize}

\item Equivalence Class Expression ([= =]) [TRIO]\\
Locale dependent (LC_COLLATE).
Only one expression can appear inside the delimiters.
\begin{description}
\item[=a=] All letters in the same equivalence class as the letter 'a'.
\end{description}
\begin{enumerate}
\item[] \texttt{trio_scanf("\%[[=a=]b]\Bs n", buffer);}
\item[] \texttt{trio_scanf("\%[[=a=][=b=]]\Bs n", buffer);}
\end{enumerate}

\item Character Class Expression ([: :]) [TRIO]\\
Locale dependent (LC_CTYPE).
Only one expression can appear inside the delimiters.
\begin{description}
\item[:alnum:] :alpha: and :digit:
\item[:alpha:] :lower: and :upper:
\item[:cntrl:] Control characters
\item[:digit:] Decimal digits
\item[:graph:] Printable characters except space
\item[:lower:] Lower case alphabetic letters
\item[:print:] Printable characters
\item[:punct:] Punctuation
\item[:space:] Whitespace characters
\item[:upper:] Upper case alphabetic letters
\item[:xdigit:] Hexadecimal digits
\end{description}
\begin{enumerate}
\item[] \texttt{trio_scanf("\%[[:alnum:]]\Bs n", buffer);}
\item[] \texttt{trio_scanf("\%[[:alpha:][:digit:]]\Bs n", buffer);}
\end{enumerate}

\end{description}

\end{description}
